##### Extra Settings

```
// PHP
"php.suggest.basic": false,
"php.validate.enable": false,
"twig-language-2.formatting": false,

"editor.quickSuggestions": {
    "comments": true
},

"emmet.includeLanguages": {
    "twig": "html"
},

"files.associations": {
    "*.html": "twig"
},

"intelephense.files.exclude": [
    "**/.git/**",
    "**/.svn/**",
    "**/.hg/**",
    "**/CVS/**",
    "**/.DS_Store/**",
    "**/node_modules/**",
    "**/bower_components/**"
]
```
